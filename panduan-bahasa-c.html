#include — "Izin Pakai Fitur"

Sebelum program C bisa melakukan apapun, kamu harus kasih tahu "library" apa yang mau dipakai. Bayangkan seperti memilih alat sebelum masak — kalau mau goreng, ambil wajan dulu.

contoh-include.c 
#include <stdio.h>   // untuk printf dan scanf
#include <string.h>  // untuk strcpy, strlen, dll
#include <math.h>    // untuk sqrt, pow, dll
#include <stdlib.h>  // untuk rand, exit, dll
#include <stdio.h>Yang paling wajib ada di setiap program C. Tanpa ini, printf dan scanf tidak bisa dipakai.
#include <string.h>Diperlukan kalau mau pakai fungsi teks seperti strcpy (salin teks) atau strlen (hitung panjang teks).
#include <math.h>Diperlukan kalau mau pakai fungsi matematika seperti sqrt() (akar kuadrat) atau pow() (pangkat).
Untuk pemula, #include <stdio.h> hampir selalu wajib ada. Yang lain ditambahkan sesuai kebutuhan program.

Tipe Data — "Jenis Wadah"

Sebelum menyimpan data, kamu harus tentukan dulu jenis datanya. Seperti memilih wadah yang tepat — air pakai gelas, beras pakai karung.

TipeUntuk ApaContoh IsiContoh Deklarasi

floatAngka desimal3.14, 90.5, -1.5float nilai = 87.5;

int main() {

Format Specifier — "Kode Jenis Data"

Saat pakai printf atau scanf, kamu perlu kasih tahu "data ini jenisnya apa" pakai kode khusus yang disebut format specifier.

KodeUntuk TipeContoh PakaiHasilnya
%dint (angka bulat)printf("%d", 42);42
%ffloat / doubleprintf("%f", 3.14);3.140000
%.2ffloat, 2 desimalprintf("%.2f", 3.14159);3.14
%schar[] (teks)printf("%s", nama);Budi
%cchar (1 karakter)printf("%c", 'A');A
%lfdouble (di scanf)scanf("%lf", &x);—
Perbedaan penting: %.2f hanya berlaku di printf (untuk tampilan). Di scanf, selalu pakai %f saja — kalau pakai %.2f di scanf, hasilnya bisa error atau -nan!

printf & scanf — "Ngomong & Dengerin"

printf = program ngomong ke user (tampilkan teks di layar). scanf = program dengerin input dari user (baca yang diketik).

printf-scanf.c 
#include <stdio.h>

int main() {
    char nama[50];
    int  umur;
    float nilai;

    printf("Masukkan nama  : ");
    scanf("%s", nama);       // teks: tidak perlu &

    printf("Masukkan umur  : ");
    scanf("%d", &umur);      // angka: wajib pakai &

    printf("Masukkan nilai : ");
    scanf("%f", &nilai);     // angka desimal: wajib pakai &

    printf("\nHalo, %s!\n", nama);
    printf("Umur  : %d tahun\n", umur);
    printf("Nilai : %.2f\n", nilai);

    return 0;
}
&umurTanda & di scanf artinya "simpan ke alamat variabel ini". Wajib untuk int, float, double. Khusus char[] (teks) tidak perlu karena array sudah otomatis berupa alamat.
\nKarakter newline — artinya pindah ke baris baru. Tanpa ini semua output akan nempel di satu baris.
Output (contoh input: Budi, 20, 87.5)
Masukkan nama : Budi Masukkan umur : 20 Masukkan nilai : 87.5 Halo, Budi! Umur : 20 tahun Nilai : 87.50

if / else — "Kalau... Kalau Tidak..."

Percabangan — program memilih jalur yang berbeda tergantung kondisi. Seperti persimpangan jalan: kalau hujan bawa payung, kalau tidak ya sudah.

ifelse.c 
#include <stdio.h>

int main() {
    float nilai;

    printf("Masukkan nilai : ");
    scanf("%f", &nilai);

    if (nilai >= 85) {
        printf("Grade : A\n");
    } else if (nilai >= 75) {
        printf("Grade : B\n");
    } else if (nilai >= 65) {
        printf("Grade : C\n");
    } else if (nilai >= 55) {
        printf("Grade : D\n");
    } else {
        printf("Grade : E\n");
    }

    return 0;
}
if (kondisi)Cek kondisi pertama. Kalau benar, jalankan isinya dan skip semua else di bawahnya.
else ifCek kondisi berikutnya, tapi hanya kalau kondisi sebelumnya salah. Bisa ditulis sebanyak yang dibutuhkan.
elseDijalankan kalau SEMUA kondisi di atas salah. Tidak perlu kondisi, otomatis jadi "sisa dari semuanya".
Output (contoh input: 78)
Masukkan nilai : 78 Grade : B

switch / case — "Menu Pilihan"

Alternatif dari if/else yang lebih rapi kalau pilihannya banyak dan berupa angka atau karakter tertentu. Cocok banget untuk bikin menu program.

switch.c 
#include <stdio.h>

int main() {
    int pilihan;

    printf("=== Menu ===\n");
    printf("1. Tambah Data\n");
    printf("2. Lihat Data\n");
    printf("3. Keluar\n");
    printf("Pilih : ");
    scanf("%d", &pilihan);

    switch (pilihan) {
        case 1:
            printf("Tambah data dipilih\n");
            break;
        case 2:
            printf("Lihat data dipilih\n");
            break;
        case 3:
            printf("Sampai jumpa!\n");
            break;
        default:
            printf("Pilihan tidak valid!\n");
    }

    return 0;
}
switch(pilihan)Cek nilai dari variabel pilihan, lalu lompat ke case yang cocok.
break;Wajib ada di setiap case! Tanpa break, program akan lanjut terus ke case berikutnya meskipun sudah cocok.
default:Seperti else — dijalankan kalau tidak ada case yang cocok.
Output (contoh input: 2)
=== Menu === 1. Tambah Data 2. Lihat Data 3. Keluar Pilih : 2 Lihat data dipilih

Loop — "Ulang Terus Sampai Selesai"

Loop dipakai kalau ada perintah yang perlu diulang berkali-kali. Ada tiga jenis: for, while, dan do while.

loop.c 
#include <stdio.h>

int main() {
    int i;

    // FOR — kalau tahu berapa kali mau diulang
    for (i = 1; i <= 3; i++) {
        printf("for ke-%d\n", i);
    }

    // WHILE — cek kondisi dulu, baru jalan
    i = 1;
    while (i <= 3) {
        printf("while ke-%d\n", i);
        i++;
    }

    // DO WHILE — jalan dulu, baru cek kondisi (minimal 1x jalan)
    i = 1;
    do {
        printf("do-while ke-%d\n", i);
        i++;
    } while (i <= 3);

    return 0;
}
for(i=1; i<=3; i++)Baca: "mulai dari i=1, selama i kurang dari atau sama dengan 3, tambah i setiap putaran". Cocok kalau jumlah pengulangannya sudah pasti.
while(kondisi)Cek kondisi dulu sebelum jalan. Kalau kondisi langsung salah dari awal, isi loop tidak akan pernah dijalankan.
do { } while(kondisi)Jalan dulu minimal 1 kali, baru cek kondisi. Cocok untuk menu — menu pasti tampil minimal sekali sebelum dicek pilihannya.
Output
for ke-1 for ke-2 for ke-3 while ke-1 while ke-2 while ke-3 do-while ke-1 do-while ke-2 do-while ke-3

Array — "Banyak Variabel dalam Satu Nama"

Daripada bikin nilai1, nilai2, nilai3... lebih praktis pakai array. Bayangkan seperti rak dengan nomor slot — satu nama, banyak isi.

array.c 
#include <stdio.h>

int main() {
    float nilai[4];   // 4 kotak untuk menyimpan nilai
    int i;
    float total = 0;

    // Input 4 nilai
    for (i = 0; i < 4; i++) {
        printf("Nilai ke-%d : ", i+1);
        scanf("%f", &nilai[i]);
        total += nilai[i];
    }

    printf("Rata-rata : %.2f\n", total / 4);

    return 0;
}
nilai[4]Array dengan 4 slot. Nomornya mulai dari nilai[0] sampai nilai[3] — bukan dari 1! Ini kesalahan paling umum.
nilai[i]Akses slot ke-i dari array. Saat loop berjalan, i berubah dari 0 ke 3 — jadi semua slot diisi satu per satu.
total += nilai[i]Singkatan dari total = total + nilai[i] — tambahkan nilai ke total setiap putaran loop.
Output (contoh input: 80, 90, 75, 85)
Nilai ke-1 : 80 Nilai ke-2 : 90 Nilai ke-3 : 75 Nilai ke-4 : 85 Rata-rata : 82.50

String & strcpy — "Teks di Bahasa C"

C tidak punya tipe data "string" seperti bahasa lain. Teks disimpan sebagai deretan karakter (char[]). Karena itu, mengisi teks tidak bisa pakai tanda = biasa — harus pakai strcpy.

string.c 
#include <stdio.h>
#include <string.h>   // wajib untuk strcpy

int main() {
    char status[20];
    float nilai = 80;

    // status = "Lulus";  ← ini SALAH, tidak bisa!
    strcpy(status, "Lulus");   // ← ini yang benar

    printf("Status : %s\n", status);
    printf("Panjang: %d karakter\n", (int)strlen(status));

    return 0;
}
strcpy(tujuan, sumber)Menyalin teks dari sumber ke tujuan, karakter per karakter. Butuh #include <string.h>.
strlen(teks)Menghitung panjang teks (jumlah karakter). "Lulus" → 5 karakter.
Alternatif lebih simpel: daripada pakai strcpy, langsung printf("Status : Lulus\n") di dalam blok if/else — tidak perlu variabel status sama sekali!
Output
Status : Lulus Panjang: 5 karakter

File I/O — "Simpan & Baca dari File"

Supaya data tidak hilang saat program ditutup, kita bisa simpan ke file .txt. Ada 3 langkah: buka file → tulis/baca → tutup file.

Mode fopenArtinyaKalau file belum ada
"w"Write — tulis dari awal (timpa isi lama)Dibuat otomatis
"a"Append — tambah di bawah data lamaDibuat otomatis
"r"Read — baca isi fileGagal (NULL)
file-io.c 
#include <stdio.h>

int main() {
    char nama[50] = "Budi";
    float nilai = 87.5;

    // === SIMPAN KE FILE ===
    FILE *file = fopen("data.txt", "a");
    fprintf(file, "Nama  : %s\n", nama);
    fprintf(file, "Nilai : %.2f\n", nilai);
    fprintf(file, "-----\n");
    fclose(file);
    printf("Data tersimpan!\n");

    // === BACA DARI FILE ===
    FILE *baca = fopen("data.txt", "r");
    if (baca == NULL) {
        printf("File tidak ditemukan!\n");
    } else {
        char baris[100];
        while (fgets(baris, 100, baca) != NULL) {
            printf("%s", baris);
        }
        fclose(baca);
    }

    return 0;
}
FILE *fileVariabel khusus untuk menyimpan "akses" ke file. Tipe datanya adalah FILE (huruf besar semua).
fprintf(file, ...)Sama persis dengan printf, tapi ditulis ke file bukan ke layar. Parameter pertama selalu nama variabel file-nya.
fclose(file)Wajib! Menutup file setelah selesai. Kalau lupa, data bisa tidak tersimpan dengan benar.
fgets(baris, 100, baca)Baca satu baris dari file, simpan ke variabel baris. Diulang terus sampai file habis (sampai dapat NULL).
Output
Data tersimpan! Nama : Budi Nilai : 87.50 -----

Struct — "Wadah Data yang Berbeda Jenis"

Struct memungkinkan kamu menyimpan beberapa data berbeda jenis dalam satu "paket". Misalnya: data mahasiswa punya nama (teks), NIM (teks), dan nilai (angka) — semuanya bisa disimpan dalam satu struct.

struct.c 
#include <stdio.h>

// Definisi struct — tulis di luar main()
struct Mahasiswa {
    char  nama[50];
    char  nim[20];
    float nilai;
};

int main() {
    // Buat variabel bertipe struct Mahasiswa
    struct Mahasiswa mhs;

    printf("Nama  : ");
    scanf("%s", mhs.nama);

    printf("NIM   : ");
    scanf("%s", mhs.nim);

    printf("Nilai : ");
    scanf("%f", &mhs.nilai);

    printf("\n--- Data ---\n");
    printf("Nama  : %s\n", mhs.nama);
    printf("NIM   : %s\n", mhs.nim);
    printf("Nilai : %.2f\n", mhs.nilai);

    return 0;
}
struct Mahasiswa { }Mendefinisikan "cetakan" struct — ditulis di luar main(). Isinya adalah variabel-variabel yang dimiliki struct ini.
struct Mahasiswa mhsMembuat variabel mhs bertipe struct Mahasiswa. Baru setelah ini kamu bisa pakai mhs.nama, mhs.nilai, dll.
mhs.namaAkses isi struct pakai titik (.). Baca: "nama yang ada di dalam mhs".
Output (contoh input: Budi, 2024001, 88)
Nama : Budi NIM : 2024001 Nilai : 88 --- Data --- Nama : Budi NIM : 2024001 Nilai : 88.00

Fungsi — "Pisahkan Kode jadi Bagian-bagian"

Fungsi adalah blok kode yang bisa dipanggil berkali-kali dari mana saja. Gunanya untuk memecah program panjang jadi bagian-bagian kecil yang lebih mudah dibaca dan dipahami.

fungsi.c 
#include <stdio.h>

// Fungsi tanpa nilai balik (void)
void sapa(char nama[]) {
    printf("Halo, %s!\n", nama);
}

// Fungsi dengan nilai balik (return)
float hitungRata(float a, float b) {
    return (a + b) / 2;
}

int main() {
    sapa("Budi");                         // panggil fungsi sapa
    sapa("Ani");                          // bisa dipanggil lagi

    float hasil = hitungRata(80, 90);   // panggil dan simpan hasilnya
    printf("Rata-rata : %.1f\n", hasil);

    return 0;
}
void sapa(...)void artinya fungsi ini tidak mengembalikan nilai — dia cukup menjalankan perintah di dalamnya saja.
float hitungRata(...)Tipe float di depan artinya fungsi ini akan mengembalikan angka desimal. Wajib ada return di dalamnya.
return (a + b) / 2Mengembalikan hasil perhitungan ke tempat fungsi dipanggil. Nilai ini yang tersimpan di variabel hasil.
Output
Halo, Budi! Halo, Ani! Rata-rata : 85.0

scanf & Input Spasi — "Baca Nama Lebih dari Satu Kata"

scanf("%s") berhenti baca input saat ketemu spasi — jadi kalau kamu ketik "Budi Santoso", yang terbaca cuma "Budi". Untuk baca teks yang mengandung spasi, kamu perlu format khusus.

perbandingan-scanf.c 
#include <stdio.h>

int main() {
    char nama1[50];
    char nama2[50];

    // Cara lama — berhenti di spasi
    printf("Nama (cara lama) : ");
    scanf("%s", nama1);
    printf("Hasil : %s\n", nama1);

    // Cara baru — baca sampai Enter termasuk spasi
    printf("Nama (cara baru)  : ");
    scanf(" %[^\n]", nama2);
    printf("Hasil : %s\n", nama2);

    return 0;
}
Output (input: "Budi Santoso" untuk keduanya)
Nama (cara lama) : Budi Santoso Hasil : Budi ← "Santoso" hilang! Nama (cara baru) : Budi Santoso Hasil : Budi Santoso ← terbaca semua ✓
%[^\n]Format khusus: baca semua karakter sampai ketemu Enter (\n). Tanda ^ artinya "kecuali" — jadi artinya "baca semua kecuali newline".
" %[^\n]"Spasi di depan %[^\n] itu penting! Fungsinya untuk membuang sisa karakter Enter dari input sebelumnya. Tanpa spasi ini, scanf bisa langsung skip dan tidak baca apapun.
%sTetap dipakai untuk input yang memang satu kata, seperti NIM atau kode — tidak perlu diganti.
Kapan pakai mana? Pakai %[^\n] khusus untuk teks yang mungkin mengandung spasi (nama lengkap, nama jurusan, alamat). Untuk NIM, angka, atau kata tunggal tetap pakai %s atau %d seperti biasa.
contoh-penggunaan.c 
#include <stdio.h>

int main() {
    char nama[50];
    char nim[20];
    char jurusan[50];
    float nilai;

    printf("Nama     : ");
    scanf(" %[^\n]", nama);      // nama bisa lebih dari 1 kata

    printf("NIM      : ");
    scanf(" %[^\n]", nim);       // NIM bisa ada tanda strip dll

    printf("Jurusan  : ");
    scanf(" %[^\n]", jurusan);   // jurusan bisa lebih dari 1 kata

    printf("Nilai    : ");
    scanf("%f", &nilai);          // angka tetap pakai %f biasa

    printf("\n--- Data ---\n");
    printf("Nama    : %s\n", nama);
    printf("NIM     : %s\n", nim);
    printf("Jurusan : %s\n", jurusan);
    printf("Nilai   : %.2f\n", nilai);

    return 0;
}
Output (contoh input nama 3 kata)
Nama : Budi Santoso Nugroho NIM : 2024-001 Jurusan : Teknik Informatika Nilai : 88.5 --- Data --- Nama : Budi Santoso Nugroho NIM : 2024-001 Jurusan : Teknik Informatika Nilai : 88.50
Ringkasan perbedaan: %s = baca sampai spasi. %[^\n] = baca sampai Enter. Spasi di depan " %[^\n]" = buang sisa Enter dari input sebelumnya.

Operator — "Simbol untuk Perhitungan & Perbandingan"

Operator adalah simbol yang melakukan operasi tertentu. Ada tiga kelompok utama yang wajib kamu tahu: aritmatika, perbandingan, dan logika.

KelompokOperatorArtinyaContohHasil
Aritmatika+Penjumlahan5 + 38
-Pengurangan5 - 32
*Perkalian5 * 315
/Pembagian10 / 42 (bulat!)
%Sisa bagi (modulo)10 % 31
Perbandingan==Sama dengana == b1 (true) / 0 (false)
!=Tidak sama dengana != b1 / 0
>Lebih besara > b1 / 0
<Lebih kecila < b1 / 0
>=Lebih besar atau samaa >= b1 / 0
Logika&&DAN (kedua harus true)a > 0 && b > 01 / 0
||ATAU (salah satu true)a > 0 || b > 01 / 0
!BUKAN / kebalikan!(a > 0)1 / 0
Penugasan+=Tambah lalu simpana += 5sama dengan a = a + 5
++Tambah 1a++sama dengan a = a + 1
operator.c 
#include <stdio.h>

int main() {
    int a = 10, b = 3;

    // Aritmatika
    printf("10 + 3  = %d\n", a + b);   // 13
    printf("10 - 3  = %d\n", a - b);   // 7
    printf("10 * 3  = %d\n", a * b);   // 30
    printf("10 / 3  = %d\n", a / b);   // 3 (bukan 3.33!)
    printf("10 %% 3 = %d\n", a % b);   // 1 (sisa bagi)

    // Perbandingan — hasilnya 1 (true) atau 0 (false)
    printf("10 == 3 : %d\n", a == b);  // 0 (false)
    printf("10 != 3 : %d\n", a != b);  // 1 (true)
    printf("10 > 3  : %d\n", a > b);   // 1 (true)

    // Logika — gabungan kondisi
    printf("a>0 && b>0 : %d\n", a>0 && b>0);  // 1 (keduanya > 0)
    printf("a>50||b>0 : %d\n", a>50 || b>0); // 1 (b>0 true)

    return 0;
}
10 / 3 = 3Pembagian dua int hasilnya juga int — desimalnya dipotong! Kalau mau hasil desimal, salah satunya harus float: 10.0 / 3 = 3.333.
10 % 3 = 1Modulo = sisa bagi. 10 dibagi 3 = 3 sisa 1, jadi hasilnya 1. Berguna untuk cek genap/ganjil: angka % 2 == 0 artinya genap.
&&DAN — kedua kondisi harus benar. Contoh: nilai >= 0 && nilai <= 100 artinya nilai harus antara 0 dan 100.
||ATAU — cukup salah satu yang benar. Contoh: pilihan == 1 || pilihan == 2 artinya pilihan boleh 1 atau 2.
Output
10 + 3 = 13 10 - 3 = 7 10 * 3 = 30 10 / 3 = 3 10 % 3 = 1 10 == 3 : 0 10 != 3 : 1 10 > 3 : 1 a>0 && b>0 : 1 a>50||b>0 : 1

Array 2 Dimensi — "Tabel di dalam Program"

Array biasa seperti satu baris kotak. Array 2 dimensi seperti tabel — punya baris dan kolom. Kamu sudah pakai ini di program: char matkul[4][30] adalah tabel 4 baris, tiap baris muat 30 karakter.

array2d.c 
#include <stdio.h>

int main() {
    // Array 2D angka — 2 baris, 3 kolom
    int nilai[2][3] = {
        {80, 90, 75},   // baris 0: nilai siswa A
        {85, 70, 95}    // baris 1: nilai siswa B
    };

    // Array 2D teks — 4 nama, masing-masing max 30 karakter
    char matkul[4][30] = {
        "Aljabar Linear",
        "Pemrograman",
        "Bahasa Indonesia",
        "Kewirausahaan"
    };

    // Akses elemen tertentu
    printf("Nilai siswa A matkul ke-2 : %d\n", nilai[0][1]);
    printf("Matkul ke-3               : %s\n", matkul[2]);

    // Loop untuk tampilkan semua matkul
    int i;
    for (i = 0; i < 4; i++) {
        printf("Matkul %d : %s\n", i+1, matkul[i]);
    }

    return 0;
}
nilai[2][3]Angka pertama = jumlah baris, angka kedua = jumlah kolom. Bayangkan seperti tabel Excel 2 baris × 3 kolom.
nilai[0][1]Akses baris ke-0, kolom ke-1. Ingat: indeks mulai dari 0! Jadi baris ke-1 = indeks 0, kolom ke-2 = indeks 1.
char matkul[4][30]Array teks 2D — 4 slot teks, tiap teks maksimal 30 karakter. Inilah yang dipakai di program data mahasiswamu!
matkul[i]Untuk array teks 2D, akses satu baris teks cukup pakai satu indeks saja. Beda dengan array angka yang perlu dua indeks.
Output

for vs while vs do-while — "Kapan Pakai yang Mana?"

Ketiganya sama-sama untuk perulangan, tapi punya situasi ideal masing-masing. Salah pilih tidak selalu salah, tapi ada yang lebih cocok untuk tiap kasus.

LoopCocok untukCek kondisiMinimal jalan
forTahu pasti berapa kali diulangSebelum jalanBisa 0 kali
whileTidak tahu pasti berapa kali, tapi ada kondisi berhentiSebelum jalanBisa 0 kali
do-whileHarus jalan minimal 1 kali (seperti menu)Setelah jalanMinimal 1 kali
perbandingan-loop.c 
#include <stdio.h>

int main() {
    int i;

    // FOR — input 4 nilai matkul (tahu pasti 4 kali)
    for (i = 0; i < 4; i++) {
        printf("Input nilai matkul ke-%d\n", i+1);
    }

    // WHILE — terus minta input sampai benar (tidak tahu berapa kali)
    float nilai = -1;
    while (nilai < 0 || nilai > 100) {
        printf("Masukkan nilai (0-100): ");
        scanf("%f", &nilai);
    }

    // DO-WHILE — menu utama (harus tampil minimal 1 kali)
    int pilihan;
    do {
        printf("1. Input  2. Lihat  3. Keluar\n");
        scanf("%d", &pilihan);
    } while (pilihan != 3);

    return 0;
}
forPaling ringkas kalau jumlah pengulangannya sudah pasti. Variabel penghitung (i), kondisi, dan penambahan ditulis semua di satu baris.
whileCocok kalau kamu tidak tahu pasti berapa kali diulang — program terus jalan selama kondisi masih benar. Hati-hati infinite loop kalau kondisinya tidak pernah jadi false!
do-whileSatu-satunya loop yang dijamin jalan minimal 1 kali sebelum kondisi dicek. Paling cocok untuk menu — menu harus tampil dulu baru user bisa pilih.
Tips mudah: Kalau tahu berapa kali → pakai for. Kalau tidak tahu berapa kali tapi ada syarat berhenti → pakai while. Kalau harus jalan dulu minimal sekali → pakai do-while.

Validasi Input — "Pastikan User Input yang Benar"

Tanpa validasi, kalau user input nilai 999 atau angka negatif, program tetap jalan dan hasilnya salah. Validasi = program menolak input yang tidak masuk akal dan minta user input ulang.

validasi.c 
#include <stdio.h>

int main() {
    float nilai;
    int pilihan;

    // Validasi nilai 0-100
    do {
        printf("Masukkan nilai (0-100) : ");
        scanf("%f", &nilai);

        if (nilai < 0 || nilai > 100) {
            printf("Nilai harus antara 0 dan 100!\n");
        }
    } while (nilai < 0 || nilai > 100);

    printf("Nilai diterima: %.2f\n", nilai);

    // Validasi pilihan menu
    do {
        printf("Pilih menu (1/2/3) : ");
        scanf("%d", &pilihan);

        if (pilihan < 1 || pilihan > 3) {
            printf("Pilihan tidak valid, coba lagi!\n");
        }
    } while (pilihan < 1 || pilihan > 3);

    printf("Pilihan diterima: %d\n", pilihan);

    return 0;
}
do { } while (kondisi salah)Pola validasi paling umum — minta input dulu, cek apakah salah, kalau salah ulang terus sampai inputnya benar.
nilai < 0 || nilai > 100Kondisi "input salah" — nilai di bawah 0 ATAU di atas 100. Selama salah satu terpenuhi, loop terus berulang.
if di dalam do-whileTampilkan pesan error di dalam loop — biar user tahu kenapa inputnya ditolak sebelum diminta input ulang.
Output (user sengaja salah dulu)
Masukkan nilai (0-100) : 150 Nilai harus antara 0 dan 100! Masukkan nilai (0-100) : -5 Nilai harus antara 0 dan 100! Masukkan nilai (0-100) : 85 Nilai diterima: 85.00 Pilih menu (1/2/3) : 9 Pilihan tidak valid, coba lagi! Pilih menu (1/2/3) : 2 Pilihan diterima: 2

Pointer — "Alamat dari Variabel"

Setiap variabel di C punya alamat di memori komputer — seperti nomor rumah. Pointer adalah variabel yang isinya bukan nilai biasa, tapi alamat dari variabel lain. Kamu sebenarnya sudah pakai konsep ini tanpa sadar lewat tanda & di scanf!

pointer.c 
#include <stdio.h>

int main() {
    int nilai = 85;

    // & = ambil alamat variabel
    printf("Nilai          : %d\n",  nilai);
    printf("Alamat nilai   : %p\n", &nilai);

    // Pointer — variabel yang menyimpan alamat
    int *ptr = &nilai;   // ptr menyimpan alamat dari nilai

    printf("Alamat di ptr  : %p\n",  ptr);   // sama dengan &nilai
    printf("Isi via ptr    : %d\n", *ptr);   // *ptr = baca isi di alamat itu

    // Ubah nilai lewat pointer
    *ptr = 90;
    printf("Nilai sekarang : %d\n", nilai);   // ikut berubah jadi 90!

    return 0;
}
&nilaiTanda & = "ambil alamat". Inilah yang selalu kamu tulis di scanf("%d", &nilai) — scanf butuh tahu alamatnya biar bisa langsung nulis nilai ke sana.
int *ptrDeklarasi pointer — tanda * di depan nama variabel artinya "variabel ini adalah pointer (penyimpan alamat)" bertipe int.
*ptrTanda * saat dipakai (bukan deklarasi) artinya "buka isi di alamat itu" — disebut dereferencing. Jadi *ptr = isi dari variabel yang alamatnya disimpan ptr.
*ptr = 90Mengubah nilai lewat pointer — karena ptr menyimpan alamat variabel nilai, mengubah *ptr sama dengan mengubah nilai langsung.
Jangan bingung antara * saat deklarasi dan saat pakai! int *ptr = "ptr adalah pointer". *ptr = "isi dari alamat yang disimpan ptr". Sama simbolnya, beda artinya.
Output (alamat bisa beda di tiap komputer)
Nilai : 85 Alamat nilai : 0x7ffd5e2a1b4c Alamat di ptr : 0x7ffd5e2a1b4c Isi via ptr : 85 Nilai sekarang : 90
Untuk sekarang, yang paling penting dipahami adalah kenapa scanf butuh tanda & — karena scanf perlu alamat variabel untuk bisa menyimpan nilai ke sana. Itu sudah 80% dari kegunaan pointer di level pemula!

Nested Loop — "Loop di dalam Loop"

Nested loop = loop yang ada di dalam loop lain. Loop luar jalan 1 kali, loop dalam jalan sampai selesai — baru loop luar lanjut ke putaran berikutnya. Sering jadi soal ujian dalam bentuk pola bintang atau tabel.

nested-loop.c 
#include <stdio.h>

int main() {
    int i, j;

    // Contoh 1: Pola bintang segitiga
    printf("=== Pola Bintang ===\n");
    for (i = 1; i <= 4; i++) {      // loop luar: baris 1-4
        for (j = 1; j <= i; j++) {  // loop dalam: cetak bintang sebanyak i
            printf("* ");
        }
        printf("\n");  // pindah baris setelah loop dalam selesai
    }

    // Contoh 2: Tabel perkalian 3x3
    printf("\n=== Tabel Perkalian ===\n");
    for (i = 1; i <= 3; i++) {
        for (j = 1; j <= 3; j++) {
            printf("%dx%d=%d\t", i, j, i*j);
        }
        printf("\n");
    }

    return 0;
}
for (i ...) { for (j ...) }Loop luar pakai i, loop dalam pakai j. Tiap 1 putaran i, loop j jalan sampai habis dulu. Total putaran = luar × dalam.
j <= iKondisi loop dalam bisa pakai variabel loop luar! Di contoh pola bintang, baris ke-3 cetak 3 bintang karena j sampai i (=3).
\tTab character — bikin jarak antar kolom di tabel jadi rapi otomatis tanpa harus hitung spasi manual.
printf("\n") setelah loop dalamWajib ada di luar loop dalam tapi masih di dalam loop luar — supaya pindah baris setelah satu baris penuh dicetak.
Output
=== Pola Bintang === * * * * * * * * * * === Tabel Perkalian === 1x1=1 1x2=2 1x3=3 2x1=2 2x2=4 2x3=6 3x1=3 3x2=6 3x3=9
Cara baca nested loop: "untuk setiap 1 putaran i, j jalan dari awal sampai habis". Kalau luar 4 kali dan dalam 4 kali, total putaran = 4 × 4 = 16 kali.

break & continue — "Keluar atau Skip"

Dua kata kunci ini mengontrol jalannya loop. break = keluar total dari loop. continue = skip putaran ini, lanjut ke putaran berikutnya.

break-continue.c 
#include <stdio.h>

int main() {
    int i;

    // BREAK — berhenti saat ketemu 4
    printf("=== break ===\n");
    for (i = 1; i <= 7; i++) {
        if (i == 4) {
            break;    // keluar dari loop sepenuhnya
        }
        printf("%d ", i);
    }
    printf("\n");

    // CONTINUE — skip angka genap
    printf("=== continue ===\n");
    for (i = 1; i <= 7; i++) {
        if (i % 2 == 0) {
            continue;  // skip putaran ini, lanjut ke i berikutnya
        }
        printf("%d ", i);
    }
    printf("\n");

    return 0;
}
breakKeluar dari loop sepenuhnya — program lanjut ke baris setelah kurung kurawal loop. Di contoh, loop harusnya sampai 7 tapi berhenti di 3 karena break saat i=4.
continueSkip sisa kode di putaran ini dan langsung lompat ke putaran berikutnya. Di contoh, angka genap (2,4,6) di-skip jadi yang tercetak hanya ganjil.
i % 2 == 0Cara cek angka genap — kalau sisa bagi 2 adalah 0, berarti genap. Ini kombinasi operator modulo dan perbandingan.
Output
=== break === 1 2 3 === continue === 1 3 5 7
Ingat: break di dalam switch berbeda dengan break di dalam loop. Di switch, break keluar dari switch. Di loop, break keluar dari loop. Tapi kalau switch ada di dalam loop, break hanya keluar dari switch — bukan dari loop!

i++ vs ++i — "Tambah Dulu atau Pakai Dulu?"

Keduanya sama-sama menambah nilai variabel sebesar 1 — tapi urutannya berbeda kalau dipakai dalam ekspresi. Di loop biasa seperti for, perbedaannya tidak kelihatan. Tapi di ekspresi lain, bedanya nyata.

increment.c 
#include <stdio.h>

int main() {
    int a, b, hasil;

    // i++ : pakai dulu, baru tambah
    a = 5;
    hasil = a++;
    printf("a++  → hasil=%d, a=%d\n", hasil, a);
    // hasil = 5 (nilai a sebelum ditambah)
    // a = 6 (sudah ditambah setelahnya)

    // ++i : tambah dulu, baru pakai
    b = 5;
    hasil = ++b;
    printf("++b  → hasil=%d, b=%d\n", hasil, b);
    // hasil = 6 (nilai b setelah ditambah)
    // b = 6

    // Di loop for — tidak ada bedanya
    int i;
    for (i = 0; i < 3; i++) {  // sama saja dengan ++i di sini
        printf("i = %d\n", i);
    }

    return 0;
}
a++Post-increment — nilai a dipakai dulu dalam ekspresi, baru ditambah 1. Jadi hasil = a++ → hasil dapat nilai lama, a jadi nilai lama+1.
++bPre-increment — nilai b ditambah 1 dulu, baru dipakai. Jadi hasil = ++b → b naik dulu, hasil dapat nilai baru.
i-- dan --iBerlaku sama untuk decrement (kurang 1). i-- pakai dulu baru kurang, --i kurang dulu baru pakai.
Output
a++ → hasil=5, a=6 ++b → hasil=6, b=6 i = 0 i = 1 i = 2
Di loop for, i++ dan ++i hasilnya sama saja karena nilainya tidak langsung dipakai. Perbedaan baru terasa kalau dipakai dalam ekspresi seperti hasil = i++.

const & #define — "Nilai yang Tidak Boleh Diubah"

Konstanta adalah nilai yang ditetapkan sekali dan tidak bisa diubah selama program berjalan. Berguna untuk nilai yang tetap seperti nilai PI, batas maksimal, atau batas kelulusan — supaya tidak perlu ganti di banyak tempat kalau nilainya berubah.

konstanta.c 
#include <stdio.h>

// #define — konstanta preprocessor, ditulis di luar main
#define PI 3.14159
#define BATAS_LULUS 75
#define MAKS_SISWA 30

int main() {
    // const — konstanta variabel, ditulis di dalam main
    const float gravitasi = 9.8;
    const int   maxNilai  = 100;

    printf("PI            = %.5f\n", PI);
    printf("Batas lulus   = %d\n", BATAS_LULUS);
    printf("Gravitasi     = %.1f\n", gravitasi);
    printf("Nilai maksimal= %d\n", maxNilai);

    float nilai = 80;
    if (nilai >= BATAS_LULUS) {   // pakai konstanta, bukan angka 75
        printf("Status: Lulus\n");
    }

    // gravitasi = 10; ← ini ERROR! tidak bisa ubah const

    return 0;
}
#define NAMA nilaiPreprocessor macro — sebelum program dikompilasi, semua tulisan PI diganti dengan 3.14159 secara otomatis. Tidak butuh tipe data, tidak pakai titik koma, ditulis di luar main.
const float x = 9.8Variabel konstanta — punya tipe data, dideklarasikan seperti variabel biasa tapi ditambah kata const di depan. Kalau dicoba diubah, compiler langsung error.
HURUF_BESARKonvensi umum — nama konstanta ditulis HURUF BESAR semua supaya mudah dibedakan dari variabel biasa saat membaca kode.
Output
PI = 3.14159 Batas lulus = 75 Gravitasi = 9.8 Nilai maksimal= 100 Status: Lulus
Kapan pakai mana? #define lebih ringkas dan umum dipakai untuk nilai angka/teks sederhana. const lebih aman karena punya tipe data yang dicek compiler. Untuk pemula, keduanya boleh dipakai.

Operator Ternary — "if/else dalam Satu Baris"

Ternary adalah cara singkat menulis if/else sederhana dalam satu baris. Namanya "ternary" karena butuh 3 bagian: kondisi, nilai kalau true, nilai kalau false.

ternary.c 
#include <stdio.h>

int main() {
    float nilai = 80;
    int   umur  = 17;

    // if/else biasa — 5 baris
    char *status1;
    if (nilai >= 75) {
        status1 = "Lulus";
    } else {
        status1 = "Tidak Lulus";
    }

    // Ternary — 1 baris, hasil sama
    // kondisi ? nilai_jika_true : nilai_jika_false
    char *status2 = (nilai >= 75) ? "Lulus" : "Tidak Lulus";

    // Contoh lain
    char *kat = (umur >= 18) ? "Dewasa" : "Belum dewasa";
    int maks   = (10 > 7) ? 10 : 7;      // ambil yang lebih besar

    printf("Status  : %s\n", status2);
    printf("Kategori: %s\n", kat);
    printf("Maksimal: %d\n", maks);

    return 0;
}
kondisi ? A : BBaca: "kalau kondisi benar, hasilnya A, kalau salah hasilnya B". Tanda ? memisahkan kondisi dari nilai true, tanda : memisahkan nilai true dari false.
char *statusUntuk menyimpan hasil ternary berupa teks, pakai char * (pointer ke char) — lebih praktis daripada char[] saat nilainya langsung diisi dari ternary.
(10 > 7) ? 10 : 7Cara ringkas cari nilai terbesar dari dua angka — kalau 10 lebih besar dari 7, ambil 10, kalau tidak ambil 7.
Output
Status : Lulus Kategori: Belum dewasa Maksimal: 10
Jangan terlalu banyak ternary bersarang — a ? b ? c : d : e susah dibaca. Kalau kondisinya lebih dari satu tingkat, lebih baik tetap pakai if/else biasa.

Type Casting — "Paksa Ganti Tipe Data"

Type casting = paksa mengubah tipe data sementara untuk keperluan perhitungan. Paling sering dipakai untuk mengatasi masalah pembagian integer yang memotong desimal.

casting.c 
#include <stdio.h>

int main() {
    int a = 7, b = 2;

    // Tanpa casting — hasilnya dipotong jadi int
    printf("Tanpa casting : %d\n",   a / b);          // 3

    // Dengan casting — paksa jadi float dulu
    printf("Dengan casting: %.2f\n", (float)a / b);  // 3.50

    // Contoh nyata: hitung rata-rata
    int total = 250;
    int n     = 3;

    printf("Rata (int)  : %d\n",   total / n);              // 83 (salah!)
    printf("Rata (float): %.2f\n", (float)total / n);      // 83.33 (benar!)

    // Cast float ke int — buang desimal
    float ipk = 3.75;
    printf("IPK bulat   : %d\n",   (int)ipk);               // 3 (bukan dibulatkan!)

    return 0;
}
(float)a / bCast a jadi float sementara sebelum dibagi. Cukup satu operand yang di-cast — yang lain otomatis ikut jadi float juga.
(int)ipkCast float ke int — desimalnya langsung dipotong, bukan dibulatkan. 3.75 jadi 3, bukan 4. Kalau mau bulatkan ke atas, pakai fungsi ceil() dari math.h.
(float)total / nCara paling umum — cast pembilang jadi float supaya hasil pembagian tidak dipotong. Ini solusi untuk masalah IPK atau rata-rata yang harusnya desimal.
Output
Tanpa casting : 3 Dengan casting: 3.50 Rata (int) : 83 Rata (float): 83.33 IPK bulat : 3
Jebakan paling umum: int / int = int selalu. Jadi 1 / 2 = 0, bukan 0.5! Kalau butuh hasil desimal dari pembagian dua integer, wajib cast salah satunya ke float dulu.

Rekursi — "Fungsi yang Memanggil Dirinya Sendiri"

Rekursi adalah teknik di mana sebuah fungsi memanggil dirinya sendiri untuk menyelesaikan masalah yang bisa dipecah jadi masalah lebih kecil. Contoh klasiknya adalah faktorial dan deret Fibonacci.

rekursi.c 
#include <stdio.h>

// Fungsi faktorial rekursif
// 5! = 5 × 4 × 3 × 2 × 1 = 120
int faktorial(int n) {
    if (n == 0 || n == 1) {
        return 1;             // base case — titik berhenti
    }
    return n * faktorial(n - 1);  // panggil diri sendiri
}

// Fungsi pangkat rekursif
// pangkat(2, 4) = 2^4 = 16
int pangkat(int basis, int exp) {
    if (exp == 0) {
        return 1;             // basis^0 selalu 1
    }
    return basis * pangkat(basis, exp - 1);
}

int main() {
    printf("5!      = %d\n", faktorial(5));    // 120
    printf("0!      = %d\n", faktorial(0));    // 1
    printf("2^4     = %d\n", pangkat(2, 4));  // 16
    printf("3^3     = %d\n", pangkat(3, 3));  // 27

    return 0;
}
base caseKondisi berhenti yang wajib ada di setiap fungsi rekursif. Tanpa base case, fungsi akan terus memanggil dirinya sendiri selamanya sampai program crash.
return n * faktorial(n-1)Fungsi memanggil dirinya dengan nilai yang lebih kecil (n-1). Ini terus terjadi sampai n mencapai base case (0 atau 1).
Alur faktorial(3)faktorial(3) → 3 × faktorial(2) → 3 × 2 × faktorial(1) → 3 × 2 × 1 = 6. Fungsi "menumpuk" lalu menghitung balik dari bawah.
Output
5! = 120 0! = 1 2^4 = 16 3^3 = 27
Rekursi keren tapi boros memori dibanding loop biasa. Untuk pemula, rekursi paling sering keluar di soal faktorial atau Fibonacci. Yang wajib dipahami: selalu ada base case sebagai titik berhenti!
